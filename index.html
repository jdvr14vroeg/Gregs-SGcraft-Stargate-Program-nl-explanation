sg = peripheral.find("stargate") local param1, param2, param3 = os.pullEvent()

e = Main() -- Dit is het hoofd programa, vanaf hier kun je kiezen waar je heen gaat!

i = Iris() i1 = IrisOpen() i2 = IrisClosed() -- Dit is 1 Iris onderdeel met meerdere functies.

d = DialCheck() d1 = Idle() d2 =- Connected() -- Dit is het Dial onderdeel, hier kun je zorgen dat hij connect of disconnect.

c = SystemCheck() -- Hier kun je een systemcheck doen en kijken naar de SG state.

print("Type in the correct password in order to use the program.") print("Password:")

if read == "StargateCommand" then e else shell.reboot() end

function Main() print("Here you can choose the Function you want to use: print("[1]: Dial, [2]: Iris, [3]: SystemCheck, [4]: Disconnect Stargate") print() print("For [1] Type 1") print("For [2] Type 2") print("For [3] Type 3") print("For [4] Type 4")

if read() == "1" then d elseif read() == "2" then i elseif read() == "3" then c elseif read() == "4" then sg.disconnect() end

function DialCheck() if sg.stargateState() == "Idle" then d1 elseif sg.stargateState() == "Connected" then d2 end

function Idle() print("Type in the Stargate Address that you are trying to reach.") sg.dial(read()) sg.closeIris() sleep(20) sg.openIris() e

function Connected() print("The Stargate is in use, in order to use it you need to disconnect first.") e

function Iris() if sg.irisState() == "Opened" then i1 elseif sg.irisState() == "Closed" then i2 end

function IrisOpen() print("type close in order to close the Iris, if you do not want to close it type exit") if read() == "close" then sg.closeIris() print("Have a nice day!") e elseif read() == "exit" then print("Okay, your chose Have a nice day!") e end

function IrisClosed() print("In order to open the Iris type open, if you do not want to open it type exit") if read() == "open" then sg.openIris() print("Have a nice day!") e elseif read() == "exit" then print("Okay, your chose Have a nice day!") e end

function SystemCheck() print(sg.stargateState) print(sg.irisState) print(sg.energieState)

end end end end end end end end
